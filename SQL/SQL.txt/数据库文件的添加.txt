use fzc
go
alter database fzc add filegroup fzc2 --添加次文件组

(
name='fzc2_data',
filename='E:\数据库\练习\fzc2_data.ndf',
size=5MB,

)to filegroup fzc2 --将数据库添加到次文件组fzc2中

(
name='fzc_tab',
filename='E:\数据库\练习\fzc_tab.ldf',
size=5MB,
maxsize=100MB,
filegrowth=0
)
alter database fzc modify filegroup fzc2 default --将次文件组fzc2设置为默认文件组
