use stud_info_database
go
create view student_view
as
select st.student_id 学号,st.student_name 姓名,co.course_name 课程,sc.score 成绩
from Student_tab as st
inner join Score_tab as sc on st.student_id=sc.student_id
inner join Course_tab as co on co.course_id=sc.course_id


select * from student_view --查看视图student_view
