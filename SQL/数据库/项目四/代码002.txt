--查询成绩不及格的学生学号、课程号和成绩
select student_id as 学号,course_id as 课程名称,score as 成绩
from score_tab
where score<60

select student_id as 学号,course_id as 课程名称,score as 成绩
from score_tab
where score<=59

select student_id as 学号,course_id as 课程名称,score as 成绩
from score_tab
where score between 0 and 60

--查询100103课程的平均分、最高分、最低分，查询结果按分数降序排列平均分：
select avg(score) as 平均分
from score_tab
where course_id='100103'
最高分:
select max(score) as 最高分
from score_tab
where course_id='100103'
最低分:
select min(score) as 最低分
from score_tab
where course_id='100103'

--查询选修了两门以上课程的学生学号
select student_id as 学号
from score_tab
group by student_id
having count(student_id)>=2

--查询有两门以上的课程是85分以上的学生学号
select student_id as 学号,count(course_id) as 人数
from score_tab
where score>=85
group by student_id
having count(student_id)>=2

--查询成绩在70-89分之间的学生学号、课程号及成绩
select student_id as 学号,course_id as 课程号,score as 成绩
from score_tab
where score between 70 and 89
order by score desc

select student_id as 学号,course_id as 课程号,score as 成绩
from score_tab
where score>=70 and score<=89

--查询成绩不在70-79分之间的学生学号、课程号及成绩
select student_id as 学号,course_id as 课程号,score as 成绩
from score_tab
where score not between 70 and 79
或
select student_id as 学号,course_id as 课程号,score as 成绩
from score_tab
where score>=80 and score<=69

--查询汉族和阿昌族的学生详细信息
select * from student_tab
where nation_id in('01''02')
或
select * from student_tab
where nation_id='01' or nation_id='02'

--查询姓王，并且名为一个字的教师详细信息
select * from teacher_tab
where teacher_name like '王_'

--查询电话以15开头的学生详细信息
select * from student_tab
where tel like '15%'
